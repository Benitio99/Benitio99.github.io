147 def getEditDistance(self, originalWord, stringCompared):
148 """! Returns the edit distance between the original word and the string compared.
149 @param originalWord The original word given by the user
150 @param stringCompared The string to be compared to the original word
151 """
152 if len(originalWord) < len(stringCompared):
153 return self.getEditDistance(stringCompared, originalWord)
154 # When the originalWord is longer than the word to be compared with
155 if len(stringCompared) == 0:
156 return len(originalWord)
157 previousRow = range(len(stringCompared) + 1)
158 for originalWordindex, charOne in enumerate(originalWord):
159 currentRow = [originalWordindex + 1]
160 for comparedWordIndex, charTwo in enumerate(stringCompared):
161 insertions = previous_row[comparedWordIndex + 1] + 1
162 deletions = currentRow[comparedWordIndex] + 1
163 substitutions = previous_row[comparedWordIndex] + (charOne != charTwo)
164 currentRow.append(min(insertions, deletions, substitutions))
165 previous_row = currentRow
166 return previous_row[-1]

240 def processInput(self, input: str):
241 """! Main functionality of the program that processes the input given by the user.
242 @type input: str
243 @param input The complete input given by the user
244 """
245 if self.containsError(input):
246 # gets the word that is the error
247 self.contextError = self.extractWord(input)
248 # get word before the error and after the error
249 self.preWord, self.postWord = self.extractPreAndPost(input)
250 # replaces the error with an appropriate mask
251 self.maskedInput = self.addMask(input)
252 # get the model predictions
253 self.modelResults = self.getModelPredictions(self.maskedInput)
254 self.modelResultsSynset = []
255 # for each prediction made:
256 for prediction in self.modelResults:
257 # gets the predicted word
258 prediction = prediction.get("token_str").strip()
259 # creates a list of three words to create two bigrams for comparison
260 bigramRawInputList = []
261 bigramRawInputList.append(self.preWord)
262 bigramRawInputList.append(prediction)
263 bigramRawInputList.append(self.postWord)
264 bigramList = list(nltk.bigrams(bigramRawInputList))
265 bigramJoinedList = []
266 # joins words with a space for each bigram
267 for bigram in bigramList:
268 bigram = " ".join(bigram)
269 bigramJoinedList.append(bigram)
270 bigramFinalList = bigramJoinedList[:]
271 bigramFinalList.append(prediction)
272 # gets the synsets for each bigram in the list
273 bigramListResults = self.getSynsets(bigramFinalList)
274 # Filter out the bigrams only
275 for word in list(bigramListResults):
276 for bigram in bigramJoinedList:
277 if word == bigram:
278 bigramListResults.remove(word)
279 # appends the synsets to the full list of synsets
280 for result in bigramListResults:
281 self.modelResultsSynset.append(result)

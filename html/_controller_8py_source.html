141 def getEditDistance(self, originalWord, stringCompared):
142 """! Returns the edit distance between the original word and the string compared.
143 @param originalWord The original word given by the user
144 @param stringCompared The string to be compared to the original word
145 """
146 if len(originalWord) < len(stringCompared):
147 return self.getEditDistance(stringCompared, originalWord)
148 # When the originalWord is longer than the word to be compared with
149 if len(stringCompared) == 0:
150 return len(originalWord)
151 previousRow = range(len(stringCompared) + 1)
152 for originalWordindex, charOne in enumerate(originalWord):
153 currentRow = [originalWordindex + 1]
154 for comparedWordIndex, charTwo in enumerate(stringCompared):
155 insertions = previous_row[comparedWordIndex + 1] + 1
156 deletions = currentRow[comparedWordIndex] + 1
157 substitutions = previous_row[comparedWordIndex] + (charOne != charTwo)
158 currentRow.append(min(insertions, deletions, substitutions))
159 previous_row = currentRow
160 return previous_row[-1]

231 def processInput(self, input: str):
232 """! Main functionality of the program that processes the input given by the user.
233 @type input: str
234 @param input The complete input given by the user
235 """
236 if self.containsError(input):
237 # gets the word that is the error
238 self.contextError = self.extractWord(input)
239 # get word before the error and after the error
240 self.preWord, self.postWord = self.extractPreAndPost(input)
241 # replaces the error with an appropriate mask
242 self.maskedInput = self.addMask(input)
243 # get the model predictions
244 self.modelResults = self.getModelPredictions(self.maskedInput)
245 self.modelResultsSynset = []
246 # for each prediction made:
247 for prediction in self.modelResults:
248 # gets the predicted word
249 prediction = prediction.get("token_str").strip()
250 # creates a list of three words to create two bigrams for comparison
251 bigramRawInputList = []
252 bigramRawInputList.append(self.preWord)
253 bigramRawInputList.append(prediction)
254 bigramRawInputList.append(self.postWord)
255 bigramList = list(nltk.bigrams(bigramRawInputList))
256 bigramJoinedList = []
257 # joins words with a space for each bigram
258 for bigram in bigramList:
259 bigram = " ".join(bigram)
260 bigramJoinedList.append(bigram)
261 bigramFinalList = bigramJoinedList[:]
262 bigramFinalList.append(prediction)
263 # gets the synsets for each bigram in the list
264 bigramListResults = self.getSynsets(bigramFinalList)
265 # Filter out the bigrams only
266 for word in list(bigramListResults):
267 for bigram in bigramJoinedList:
268 if word == bigram:
269 bigramListResults.remove(word)
270 # appends the synsets to the full list of synsets
271 for result in bigramListResults:
272 self.modelResultsSynset.append(result)
273
274 # Filter out any duplicates that may be present
275 self.modelResultsSynset = list(set(self.modelResultsSynset))
276 # returns a set of lists that adds the edit distance to each word in the set
277 self.editDistances = self.getEditDistanceList()
